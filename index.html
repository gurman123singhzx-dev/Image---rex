<script>
const preview=document.getElementById("preview");
let currentFile=null;
function openCamera(){
document.getElementById("cameraInput").click();
}
function openGallery(){
document.getElementById("galleryInput").click();
}
document.getElementById("cameraInput").onchange=e=>{
currentFile=e.target.files[0];
preview.src=URL.createObjectURL(currentFile);
};
document.getElementById("galleryInput").onchange=e=>{
currentFile=e.target.files[0];
preview.src=URL.createObjectURL(currentFile);
};
function time(){
const d=new Date();
return d.getHours()+":"+String(d.getMinutes()).padStart(2,"0");
}
const chat=document.getElementById("chat");
function addMsg(t,type){
const div=document.createElement("div");
div.className="msg "+type;
div.innerHTML=t+`<span class="time">${time()}</span>`;
chat.appendChild(div);
chat.scrollTop=chat.scrollHeight;
}
function ask(){
const q=document.getElementById("question").value.trim();
if(!currentFile) return alert("Select image first");
if(!q) return alert("Type question");
addMsg("<b>You:</b> "+q,"user-msg");
addMsg("<b>AI:</b> Reading image...","ai-msg");
Tesseract.recognize(currentFile,"eng")
.then(({data:{text}})=>{
let ans="";
const l=q.toLowerCase();
if(l.includes("what")||l.includes("written")){
ans=text.trim();
}else if(l.includes("summary")){
ans=text.trim().slice(0,150)+"...";
}else{
ans=text.trim();
}
chat.lastChild.innerHTML="<b>AI:</b><br><pre>"+ans+"</pre><span class='time'>"+time()+"</span>";
})
.catch(()=>{
chat.lastChild.innerHTML="<b>AI:</b> Error reading image<span class='time'>"+time()+"</span>";
});
document.getElementById("question").value="";
}
</script>
